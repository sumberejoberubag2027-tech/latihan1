    <script>
        // Ensure scripts run after DOM is fully loaded
        document.addEventListener("DOMContentLoaded", (event) => {
            gsap.registerPlugin(ScrollTrigger);

            // 1. Initial Hero Cube Animation (Continuous Rotation)
            const cube = document.getElementById('heroCube');
            
            // Set initial state
            gsap.set(cube, { rotationX: -15, rotationY: 45 });

            // Create a continuous rotation that can be influenced by scroll
            let cubeRotation = gsap.to(cube, {
                rotationY: "+=360",
                rotationX: "+=180",
                duration: 20,
                ease: "none",
                repeat: -1,
                paused: false
            });

            // Make the cube react to scroll speed (scrub)
            ScrollTrigger.create({
                trigger: "header",
                start: "top top",
                end: "bottom top",
                scrub: 1, // Smooth scrubbing
                onUpdate: (self) => {
                    // Add extra rotation based on scroll progress
                    gsap.to(cube, {
                        rotationY: 45 + (self.progress * 360),
                        rotationX: -15 + (self.progress * 180),
                        duration: 0.5,
                        overwrite: "auto"
                    });
                }
            });

            // 2. Parallax Effects for Elements
            const parallaxItems = document.querySelectorAll('.parallax-item');
            
            parallaxItems.forEach(item => {
                const speed = item.dataset.speed || 1;
                // Calculate y movement based on speed factor
                const yMove = (1 - speed) * 100;

                gsap.fromTo(item, 
                    { y: yMove },
                    {
                        y: -yMove,
                        ease: "none",
                        scrollTrigger: {
                            trigger: item.closest('section'),
                            start: "top bottom",
                            end: "bottom top",
                            scrub: true
                        }
                    }
                );
            });

            // 3. Fade in components on scroll
            const fadeElements = document.querySelectorAll('.glass-panel');
            
            fadeElements.forEach((el, index) => {
                gsap.fromTo(el,
                    { 
                        opacity: 0, 
                        y: 50,
                        rotateX: 10
                    },
                    {
                        opacity: 1,
                        y: 0,
                        rotateX: 0,
                        duration: 1,
                        ease: "power3.out",
                        scrollTrigger: {
                            trigger: el,
                            start: "top 85%",
                            toggleActions: "play none none reverse"
                        }
                    }
                );
            });
            
            // 4. Header text reveal
            gsap.fromTo("header h2", 
                { opacity: 0, filter: "blur(10px)", scale: 0.9 },
                { opacity: 1, filter: "blur(0px)", scale: 1, duration: 1.5, stagger: 0.2, ease: "power2.out", delay: 0.5 }
            );
        });
    </script>
